## Example Use
```
import (
  "fmt"

  "github.com/aouyang1/go-forecaster"
)

func MyForecast(t []time.Time, y []float64, horizonT []time.Time) ([]float64, error) {
	f, err := forecaster.New(nil)
	if err != nil {
		return nil, err
	}
	if err := f.Fit(t, y); err != nil {
		return nil, err
	}

  return f.Predict(horizonT)
}	
```
